    . deployment

    

## service/access/hello.yaml 

apiVersion: apps/v1
kind: Deployment
metadata:
  name: hello
  namespace: backend
spec:
  selector:
    matchLabels:
      app: hello
      tier: backend
      track: stable
  replicas: 2
  template:
    metadata:
      labels:
        app: hello
        tier: backend
        track: stable
    spec:
      containers:
        - name: hello
          image: "nginx"
          ports:
            - name: http
              containerPort: 80

## service/access/hello-service.yaml 

apiVersion: v1
kind: Service
metadata:
  name: hello
  namespace: backend
spec:
  selector:
    app: hello
    tier: backend
  ports:
  - protocol: TCP
    port: 80
    targetPort: http
